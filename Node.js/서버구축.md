# 서버 구축 (http)
### - 간단한 서버 예제
```js
// require 는 타 언어 import 와 유사
// http 모듈을 http 변수에 저장하여 사용
var http = require('http'); 

// createSerer 함수로 서버 생성
// 파라미터 function(request,response) 부분은 이벤트 발생 시 콜백
// 서버가 실행됐을 때 request, response 사용 가능
var server = http.createServer(function(request,response){ 

    // writeHead (200, { '키' : '값' });
    // 200 = 정상 값 리턴 http 상태코드
    response.writeHead(200,{'Content-Type':'text/html'});
    // end() 로 실제 컨텐츠를 브라우저에 전달
    response.end('Hello node.js!!');

});

// 3333 포트로 서버 실행
server.listen(3333, function(){ 
    // 실행 확인용 로그 출력
    console.log('Server is running...');
});
```
